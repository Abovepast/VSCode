### 8位乘法（修改）
#####程序清单
``` asm
# 33H----01H
# 55H----80H
# R0 乘数 && 结果的低四位
# R1 被乘数
# R2 循环次数
# R3 部分积 && 结果的高四位

in  #用户分别输入乘数R0，被乘数R1，循环次数R2
mov R0, A
in
mov R1, A
in
mov R2, A

mov A, R0   # （06H）乘数
and A, 33H  # 与1相与判断低位
jz 0FH  # 判断乘数低位是否为0
mov A, R3   # 当低位不为零
add A, R1   # 部分积加上被乘数
jmp 10h # 低位为零，部分积不变
mov A, R3
mov 44H, A # 部份积暂存到地址44H处

mov A, 44H 
and A, 33H # 与1相与，判断部分积的低位
jz 1EH  # 低位为0则跳转
mov A, R0 # 部份积的低位为1
rr A    # 乘数右移
add A, 55H # 乘数高位补1
jmp 20H
mov A, R0 # （1EH）部份积低位为0
rr A    # 乘数右移
mov R0, A # （20H）更新乘数值

mov A, 44H  
rrc A   # 部份积右环移（带进位），不用管最低位，最低位前面已经处理过了
mov R3, A   # 更新部份积，为下一次计算准备值
#####修改部分
mov A, R2
sub A, 33H # 循环次数减1，当轮循环结束
mov R2, A
#####
jz 2DH  # 判断剩余循环次数是否为0，为0结束循环
jmp 06H # 否则继续循环
```
